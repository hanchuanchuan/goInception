# 源码安装

  • go版本v1.12及以上

  • 使用go mod做依赖管理


# 下载源码
git clone https://github.com/hanchuanchuan/goInception

cd goInception

make parser

# 构建二进制包
go build -o goInception tidb-server/main.go

# 启动(注意指定配置文件)

./goInception -config=config/config.toml
